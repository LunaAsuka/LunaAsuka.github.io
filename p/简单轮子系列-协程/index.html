void Fiber::CallerMainFunc()
{
    Fiber::ptr cur = GetThis();
    assert(cur);
    try {
        cur->m_cb();
        cur->m_cb = nullptr;
        cur->m_state = TERM;
    } catch (std::exception& ex) {
        cur->m_state = EXCEPT;
        std::cout << "Fiber Except: " << ex.what()
            << " fiber_id=" << cur->getId()
            << std::endl;

        cur->m_state = EXCEPT;
        std::cout <<  "Fiber Except"
            << " fiber_id=" << cur->getId()
            << std::endl;
    }

    auto raw_ptr = cur.get();
    cur.reset();
    raw_ptr->back();// è¿™é‡Œæ¢æˆ back
}
